} function doPost(e) {   const payload = JSON.parse(e.postData.contents);   if (payload.pw !== SECRET_PASSWORD) return;   const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Tabelle_" + payload.date);   if (sheet) {     const data = sheet.getDataRange().getValues();     const rowIndexByTime = {};     for (let r = 0; r < data.length; r++) {       rowIndexByTime[String(data[r][0])] = r;     }     // Performance: Änderungen nach Zeile bündeln, dann pro betroffener Zeile einmal schreiben.     // Achtung: slotIndex ist 1..5 (Spalten B..F), time sitzt in Spalte A.     const updatesByRow = {};     payload.changes.forEach(c => {       const rIdx = rowIndexByTime[String(c.time)];       if (rIdx === undefined) return;       if (!updatesByRow[rIdx]) updatesByRow[rIdx] = {};       updatesByRow[rIdx][c.slotIndex] = c.val;     });     Object.keys(updatesByRow)       .map(k => parseInt(k, 10))       .sort((a, b) => a - b)       .forEach(rIdx => {       // Hole aktuelle Werte der 5 Slots (B..F) und überschreibe nur betroffene Slots.       const slotRange = sheet.getRange(rIdx + 1, 2, 1, 5);       const rowSlots = slotRange.getValues(); // [[b,c,d,e,f]]       const updates = updatesByRow[rIdx];       Object.keys(updates)         .map(k => parseInt(k, 10))         .sort((a, b) => a - b)         .forEach(slotIndex => {           if (slotIndex >= 1 && slotIndex <= 5) {             rowSlots[0][slotIndex - 1] = updates[String(slotIndex)];           }         });       slotRange.setValues(rowSlots);     });     // invalidate cached day data so next load is fresh     try { CacheService.getScriptCache().remove("daydata_v1_" + payload.date); } catch (e) {}     return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON);   } }
